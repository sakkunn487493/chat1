        <div class="status">自分のIDを相手に送るか、相手のIDを入力して接続してください</div>

    </div>


    <div class="controls">

        <input type="text" id="target-id" placeholder="相手のIDを入力">

        <button onclick="connectToPeer()">相手に接続する</button>

        <hr style="border:0; border-top:1px solid #eee; margin:15px 0;">

        <input type="text" id="message-input" placeholder="メッセージを入力...">

        <button onclick="sendMessage()">送信</button>

    </div>

<script>

    let peer = new Peer(); // PeerJSのシグナリングサーバーに接続

    let conn;


    // 1. 自分のIDを取得

    peer.on('open', (id) => {

        document.getElementById('my-id').innerText = id;

    });


    // 2. 相手からの接続を待機

    peer.on('connection', (connection) => {

        conn = connection;

        setupChat();

    });


    // 3. 自分から相手に接続

    function connectToPeer() {

        const targetId = document.getElementById('target-id').value;

        if (!targetId) return alert("相手のIDを入力してください");

        

        conn = peer.connect(targetId);

        setupChat();

    }


    // 4. チャットのデータ受信設定

    function setupChat() {

        conn.on('open', () => {

            appendStatus("接続完了！メッセージを送れます");

            

            conn.on('data', (data) => {

                appendMessage(data, 'peer');

            });

        });

        

        conn.on('close', () => {

            appendStatus("切断されました");

        });

    }


    // 5. メッセージ送信

    function sendMessage() {

        const input = document.getElementById('message-input');

        const msg = input.value;

        

        if (conn && conn.open) {

            conn.send(msg);

            appendMessage(msg, 'me');

            input.value = "";

        } else {

            alert("まだ接続されていません");

        }

    }


    // 画面にメッセージを表示

    function appendMessage(text, side) {

        const log = document.getElementById('chat-log');

        const div = document.createElement('div');

        div.className = `msg ${side}`;

        div.innerText = text;

        log.appendChild(div);

        log.scrollTop = log.scrollHeight;

    }


    function appendStatus(text) {

        const log = document.getElementById('chat-log');

        const div = document.createElement('div');

        div.className = 'status';

        div.innerText = text;

        log.appendChild(div);

    }

</script>
